   15 module WordPuzzle ( remove
   16                   , isValid
   17                   ) where
   18 
   19 import qualified Data.ByteString.Char8 as Char8 (ByteString, cons, elem, empty,
   20                                                  head, tail, uncons)
   21 
   22 -- | Pattern for empty ByteString.
   23 pattern Empty :: Char8.ByteString
   24 pattern Empty <- (Char8.uncons -> Nothing)
   25 
   26 -- | Remove first occurrence of a character from a word.
   27 remove :: Char -> Char8.ByteString -> Char8.ByteString
   28 remove _ Empty = Char8.empty
   29 remove x ys = if x == h
   30                 then t
   31                 else h `Char8.cons` remove x t
   32               where
   33                 h = Char8.head ys
   34                 t = Char8.tail ys
   35 
   36 -- | Check if a word contains only characters from a list.
   37 --
   38 -- * If all valid characters are removed from the word, and there are still
   39 -- characters left over, then the word is not valid.
   40 --
   41 -- * If all valid characters are removed from the word, and the word is
   42 -- empty, then the word is valid.
   43 isValid :: String -> Char8.ByteString -> Bool
   44 isValid _  Empty  = True
   45 isValid [] _      = False
   46 isValid (x:xs) ys = if x `Char8.elem` ys
   47                       then isValid xs (x `remove` ys)
   48                       else isValid xs ys
