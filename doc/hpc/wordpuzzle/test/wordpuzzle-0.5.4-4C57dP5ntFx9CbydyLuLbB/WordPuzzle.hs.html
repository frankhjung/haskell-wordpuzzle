   13 module WordPuzzle (
   14                     remove

   19 -- | Remove first occurrence of a character from the word.
   20 --
   21 -- c.f. http://hackage.haskell.org/package/base/docs/Data-List.html#v:delete
   22 remove :: Char -> String -> String
   23 remove _ []     = []
   24 remove x (y:ys) = if x == y
   25                     then ys
   26                     else y : remove x ys
   27 
   28 -- | Check if a word contains only characters from a list.
   29 -- If all valid characters are removed from the word, and there are still
   30 -- characters left over, then the word is not valid.
   31 -- If all valid characters are removed from the word, and the word is
   32 -- empty, then the word is valid.
   33 --
   34 -- See also Data.List intersection function, '\\'
   35 isValid :: String -> String -> Bool
   36 isValid _  []     = True
   37 isValid [] _      = False
   38 isValid (x:xs) ys = if x `elem` ys
   39                       then isValid xs (x `remove` ys)
   40                       else isValid xs ys
   41 
   42 -- | Check if word is a plural.
   43 --
   44 -- This assumes the word is at least 2 characters long.
   45 --
   46 -- (1) False if word does not end in 's'
   47 -- 2. False if word ends in "ss"
   48 -- 3. True Otherwise as ends in single 's'
   49 --
   50 isPlural :: String -> Bool
   51 isPlural word
   52   | end /= 's'  = False  -- last character not an 's'
   53   | pen == "ss" = False  -- word does not end in "ss"
   54   | otherwise   = True
   55   where
   56     end = last word
   57     pen = drop (length word - 2) word
